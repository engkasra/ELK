#Generate a Certificate Authority (CA) certificate and key:
openssl req -new -x509 -sha256 -nodes -keyout ca-private.key -out ca.crt -subj "/CN=MyCA"

#Generate a private key and certificate signing request (CSR) for mahdi1:
openssl req -new -nodes -sha256 -keyout mahdi1.key -out mahdi1.csr -subj "/CN=mahdi1"

#Sign the mahdi1 certificate with the CA certificate:
openssl x509 -req -sha256 -in mahdi1.csr -CA ca.crt -CAkey ca-private.key -CAcreateserial -out mahdi1.crt

#Generate a private key and certificate signing request (CSR) for mahdi2:
openssl req -new -nodes -sha256 -keyout mahdi2.key -out mahdi2.csr -subj "/CN=mahdi2"

#Sign the mahdi2 certificate with the CA certificate:
openssl x509 -req -sha256 -in mahdi2.csr -CA ca.crt -CAkey ca-private.key -CAcreateserial -out mahdi2.crt

#Generate a private key and certificate signing request (CSR) for mahdi3:
openssl req -new -nodes -sha256 -keyout mahdi3.key -out mahdi3.csr -subj "/CN=mahdi3"

#Sign the mahdi2 certificate with the CA certificate:
openssl x509 -req -sha256 -in mahdi3.csr -CA ca.crt -CAkey ca-private.key -CAcreateserial -out mahdi3.crt



mahdi1.key: Private key for mahdi1

mahdi1.crt: Certificate signed by the CA for mahdi1

mahdi2.key: Private key for mahdi2

mahdi2.crt: Certificate signed by the CA for mahdi2

mahdi3.key: Private key for mahdi3

mahdi3.crt: Certificate signed by the CA for mahdi3
